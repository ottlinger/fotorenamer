24      private static final Logger LOG =

28       * Private default constructor to prevent instantiation of this class.

31          // prevent instantiation

34      /**
35       * Gauge the given component on screen.
36       *
37       * @param component Swing-component to gauge.
38       */
39      public static void makeCentered(final Component component) {
40          Toolkit toolkit = Toolkit.getDefaultToolkit();
41          Dimension dim = toolkit.getScreenSize();
42          int screenWidth = dim.width;
43          int screenHeight = dim.height;
44          component.setLocation((screenWidth - component.getSize().width) / 2,
45                  (screenHeight - component.getSize().height) / 2);
46      } // end of makeCentered
47  
48      /**
49       * Helper that transforms a given path into an ImageIcon. In case of errors
50       * <code>null</code> is returned.
51       *
52       * @param path File path that is to be converted into an icon.
53       * @return ImageIcon, located at the given path.
54       */
55      public static ImageIcon createImageIcon(final String path) {
56          assert path != null : "Path for image icon needs to be set.";
57          LOG.debug("Creating image icon from path " + path);
58          java.net.URL imgURL = ComponentGaugeUtil.class.getResource(path);
59          LOG.debug("extracted URL is: " + imgURL);
60  
61          if (imgURL != null) {
62              return new ImageIcon(imgURL);
63          } else {
64              LOG.error("Could not generate a valid URL from the given path: "
65                      + path);
66              return null;
67          } // end if
68      } // end of createImageIcon
